In C, strings are sequences of characters terminated by a null character ('\0'). They are commonly used to represent text data.

1. Declaration of Strings:
Strings in C are declared using character arrays. For example:

char str1[] = "Hello";
char str2[6] = {'H', 'e', 'l', 'l', 'o', '\0'};
char str3[6] = "Hello";

All three declarations create strings containing the characters "Hello", with the null character ('\0') automatically added at the end.

String Initialization:
Strings in C are typically initialized using double quotes.

#include <stdio.h>
    
    int main() {
        char str[] = "Hello, World!";
        printf("String: %s\n", str);
    
        return 0;
    }

Accessing String Characters:
Individual characters in a string can be accessed using array notation.

#include <stdio.h>
        
        int main() {
            char str[] = "Hello";
            printf("First character: %c\n", str[0]); // Outputs 'H'
        
            return 0;
        }

String Input and Output:
Strings can be read from the user using the scanf() function and printed using the printf() function.

#include <stdio.h>
            
            int main() {
                char str[50];
                printf("Enter a string: ");
                scanf("%s", str); // Input string from user
                printf("You entered: %s\n", str); // Output string
            
                return 0;
            }

1. strcpy():
Copies the contents of one string to another.

#include <stdio.h>
    #include <string.h>
    
    int main() {
        char str1[20] = "Hello";
        char str2[20];
    
        // Copy str1 to str2
        strcpy(str2, str1);
        printf("str2: %s\n", str2);
    
        return 0;
    }

2. strcat():
Concatenates two strings.

#include <stdio.h>
    #include <string.h>
    
    int main() {
        char str1[20] = "Hello";
        char str2[20] = " World";
    
        // Concatenate str2 to str1
        strcat(str1, str2);
        printf("str1: %s\n", str1);
    
        return 0;
    }

3. strcmp():
Compares two strings.

#include <stdio.h>
    #include <string.h>
    
    int main() {
        char str1[] = "Hello";
        char str2[] = "World";
    
        // Compare str1 and str2
        int result = strcmp(str1, str2);
        if (result == 0) {
            printf("Both strings are equal.\n");
        } else {
            printf("Strings are not equal.\n");
        }
    
        return 0;
    }
