Operators in C

1. Arithmetic Operators

Arithmetic operators are used to perform mathematical operations like addition, subtraction, multiplication, division, and modulus.

#include <stdio.h>

int main() {
    int a = 10, b = 5;
    int sum = a + b;       // Addition
    int difference = a - b;  // Subtraction
    int product = a * b;    // Multiplication
    int quotient = a / b;   // Division
    int remainder = a % b;  // Modulus (Remainder)
    
    printf("Sum: %d\n", sum);               // Output: Sum: 15
    printf("Difference: %d\n", difference); // Output: Difference: 5
    printf("Product: %d\n", product);       // Output: Product: 50
    printf("Quotient: %d\n", quotient);     // Output: Quotient: 2
    printf("Remainder: %d\n", remainder);   // Output: Remainder: 0
    
    return 0;
}

2. Assignment Operators

Assignment operators are used to assign values to variables.

#include <stdio.h>

int main() {
    int a = 10;
    int b;
    
    // Assignment
    b = a;
    
    printf("Value of b: %d\n", b);  // Output: Value of b: 10
    
    return 0;
}

3. Comparison Operators

Comparison operators are used to compare two values. They return true (1) if the comparison is true, and false (0) otherwise.

#include <stdio.h>

int main() {
    int a = 10, b = 20;
    
    // Comparison
    printf("a == b: %d\n", a == b);  // Output: a == b: 0 (False)
    printf("a != b: %d\n", a != b);  // Output: a != b: 1 (True)
    printf("a < b: %d\n", a < b);    // Output: a < b: 1 (True)
    printf("a > b: %d\n", a > b);    // Output: a > b: 0 (False)
    printf("a <= b: %d\n", a <= b);  // Output: a <= b: 1 (True)
    printf("a >= b: %d\n", a >= b);  // Output: a >= b: 0 (False)
    
    return 0;
}

4. Logical Operators

Logical operators are used to perform logical operations such as AND, OR, and NOT.

#include <stdio.h>

int main() {
    int a = 1, b = 0;
    
    // Logical AND
    printf("a && b: %d\n", a && b);  // Output: a && b: 0 (False)
    
    // Logical OR
    printf("a || b: %d\n", a || b);  // Output: a || b: 1 (True)
    
    // Logical NOT
    printf("!a: %d\n", !a);          // Output: !a: 0 (False)
    
    return 0;
}
