You can swap the values of two variables without using a temporary variable by using arithmetic operations or bitwise operations.

1. Using Arithmetic Operations:

#include <stdio.h>

int main() {
  int a = 5, b = 10;
  
  printf("Before swapping: a = %d, b = %d\n", a, b);
  
  a = a + b;
  b = a - b;
  a = a - b;
  
  printf("After swapping: a = %d, b = %d\n", a, b);
  
  return 0;
}

Output:

Before swapping: a = 5, b = 10
After swapping: a = 10, b = 5

2. Using Bitwise XOR Operation:

#include <stdio.h>

int main() {
  int a = 5, b = 10;
  
  printf("Before swapping: a = %d, b = %d\n", a, b);
  
  a = a ^ b;
  b = a ^ b;
  a = a ^ b;
  
  printf("After swapping: a = %d, b = %d\n", a, b);
  
  return 0;
}

Output:

Before swapping: a = 5, b = 10
After swapping: a = 10, b = 5
